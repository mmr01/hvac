cmp <- function(tm, xN,yN)
{

if (length(which(is.na(xN))) > 0)
{
    x <- xN[-which(is.na(xN))]
    y <- yN[-which(is.na(xN))]

}

else
{
    x <- xN
    y <- yN
}

if (length(which(is.na(yN))) > 0)

    xN <- x
    yN <- y

    x <- xN[-which(is.na(yN))]
    y <- yN[-which(is.na(yN))]
